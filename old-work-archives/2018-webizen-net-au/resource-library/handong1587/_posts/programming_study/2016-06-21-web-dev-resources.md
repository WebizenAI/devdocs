```

linjinbin@ubuntu:/usr/local/lib/node_modules/bower/bin$ ./bower -v

linjinbin@ubuntu:/usr/local/lib/node_modules/bower/bin$ sudo ln -s /usr/bin/nodejs /usr/bin/node
```
